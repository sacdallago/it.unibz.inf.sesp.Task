DROP TABLE IF EXISTS relation;

CREATE  TABLE task (
        id SERIAL ,
        name VARCHAR(45) NOT NULL ,
        importance INT NOT NULL ,
        duration INT NOT  NULL ,
        description VARCHAR(225) DEFAULT '<<No description provided.>>' ,
        creation TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
        status INT DEFAULT 0,
        PRIMARY KEY (id)
);

CREATE  TABLE relation (
        father INT NOT NULL,
                FOREIGN KEY (father)
                REFERENCES task (id)
                ON DELETE CASCADE
                ON UPDATE CASCADE,
        child INT NOT NULL,
                FOREIGN KEY (child)
                REFERENCES task (id)
                ON DELETE CASCADE
                ON UPDATE CASCADE,
        PRIMARY KEY (father, child)
);

DROP TABLE IF EXISTS relation;

DROP TABLE IF EXISTS auth;

CREATE  TABLE auth (
        username text NOT NULL,
        password text NOT NULL,
        PRIMARY KEY (username)
);

CREATE  TABLE task (
        id SERIAL ,
        name VARCHAR(45) NOT NULL ,
        importance INT NOT NULL ,
        duration INT NOT  NULL ,
        description VARCHAR(225) DEFAULT '<<No description provided.>>' ,
        creation TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
        status INT DEFAULT 0,
        username text NOT NULL,
                FOREIGN KEY (username)
                REFERENCES auth (username)
                ON DELETE CASCADE
                ON UPDATE CASCADE,
        PRIMARY KEY (id)
);

CREATE  TABLE relation (
        father INT NOT NULL,
                FOREIGN KEY (father)
                REFERENCES task (id)
                ON DELETE CASCADE
                ON UPDATE CASCADE,
        child INT NOT NULL,
                FOREIGN KEY (child)
                REFERENCES task (id)
                ON DELETE CASCADE
                ON UPDATE CASCADE,
        username text NOT NULL,
                FOREIGN KEY (username)
                REFERENCES auth (username)
                ON DELETE CASCADE
                ON UPDATE CASCADE,
        PRIMARY KEY (father, child, username)
);
